[%- page.import({ title => 'Sending an email with Perl', section => 'examples', }) -%] [% WRAPPER code_chunk -%] [% INCLUDE "examples/script_head.html" -%] # first, create your message use Email::MIME; my $message = Email::MIME->create( header_str => [ From => 'you@example.com', To => 'friend@example.com', Subject => 'Happy birthday!', ], attributes => { encoding => 'quoted-printable', charset => 'ISO-8859-1', }, body_str => "Happy birthday to you!\n", ); # send the message use Email::Sender::Simple qw(sendmail); sendmail($message); [%- END %]
